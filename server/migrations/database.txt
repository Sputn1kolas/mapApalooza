users {
  id interator primary key
  username varchar
  email: varchar
  password: varchar
}


maps {
  id interator primary key,
  user_id varchar references users(id),
  title varchar,
  description text,
  img_url  varchar,
  lat FLOAT(10,6),
  long FLOAT(10,6) ,
}

points {
  id interator primary key,
  map_id references map(id),
  lat FLOAT(10,6)
  long FLOAT(10,6)
  address VARCHAR ,
  title varchar,
  description text,
  img_url: VARCHAR,
}


user_favorites {
  user_id references user(id)
  map_id references map(id)
}
